N x N Solution

#include <iostream>
using namespace std;

bool isSafe(char** board,int n,int row,int col);
void print(char** board,int n);
void nqueen(char** board,int n,int row);

bool isSafe(char** board,int n,int row,int col){
//vertical up
for(int i=row-1;i>=0;i--){
     if(board[i][col]=='Q'){
     return false;
  }
}

//left diagonal

for(int i=row-1,j=col-1;i>=0 && j>=0;i--,j--){
    if(board[i][j]=='Q'){
       return false;
   }
}

//right diagonal
for(int i=row-1,j=col+1;i>=0 && j<=n;i--,j++){
      if(board[i][j]=='Q'){
      return false;
   }
}

return true;

}
void print(char** board,int n){
    for(int i=0;i<n;i++){
      for(int j=0;j<n;j++){
        cout<<board[i][j]<<" ";
      }
     cout<<endl;
   }
   cout<<endl;
}
void nqueen(char** board,int n,int row){
     if(row==n){
       print(board,n);
       return ;
}

for(int j=0;j<n;j++){
     if(isSafe(board,n,row,j)){
          board[row][j]='Q';
          nqueen(board,n,row+1);
          board[row][j]='X';
     }
}

}


int main() {
  int n;
  cin>>n;
  char** board= new char*[n];
  //Initialize
  for(int i=0;i<n;i++){
     board[i]=new char[n];
  for(int j=0;j<n;j++){
     board[i][j]='X';
  }
}

nqueen(board,n,0);

return 0;
}
